swift
class Animal {
  var name: String
  init(name: String) {
    self.name = name
  }
  func makeSound() {
      print("...")
  }
}

class Dog: Animal {
  func fetch() {
    print("\(name) is fetching!")
  }

  // йҮҚеҶҷзҲ¶зұ»ж–№жі•
  override func makeSound() {
    print("Woof!")
  }
}

let dog = Dog(name: "Buddy")
dog.makeSound()   // Woof!
dog.fetch()       // Buddy is fetching!
