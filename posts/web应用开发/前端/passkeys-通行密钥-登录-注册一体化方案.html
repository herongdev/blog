  2. иӢҘиҝ”еӣһвҖңжІЎжңүеҸҜз”ЁеҮӯиҜҒ/з”ЁжҲ·еҸ–ж¶Ҳ/NotFoundвҖқпјҡеүҚз«Ҝз«ӢеҚіи°ғз”Ё /passkeys/register-or-login/optionsпјҲеёҰдёҠ deviceIdentifierпјүгҖӮ

    • еҗҺз«Ҝ JIT е»әеҸ·пјҲжҲ–дҪҝз”ЁдҪ е·Іжңүзҡ„и®ҝе®ў/еҫ…жҝҖжҙ»иҙҰеҸ·зӯ–з•ҘпјүпјҢзӯҫеҸ‘ registration optionsпјҲresidentKey=requiredпјүгҖӮ
    • еүҚз«ҜеҸ‘иө·жіЁеҶҢд»ӘејҸ вҶ’ жӢҝеҲ° attestation еӣһ /register-or-login/verifyгҖӮ
    • жңҚеҠЎз«Ҝ ж ЎйӘҢ + е…Ҙеә“ + з»‘е®ҡи®ҫеӨҮ(еҸҜйҖү) + жҝҖжҙ»иҙҰеҸ· + зӯҫеҸ‘ JWT вҶ’ жіЁеҶҢеҚізҷ»еҪ•гҖӮ

жҳҜеҗҰеҝ…йЎ»е…ҲвҖңз»‘е®ҡеҶҚзҷ»еҪ•вҖқпјҹ дёҚеҝ…йЎ»гҖӮеҜ№е·ІеӯҳеңЁйҖҡиЎҢеҜҶй’Ҙзҡ„з”ЁжҲ·пјҢз¬¬дёҖжӯҘе°ұиғҪзҷ»еҪ•пјӣеҜ№ж–°и®ҫеӨҮ/ж–°з”ЁжҲ·пјҢз¬¬дәҢжӯҘе®ҢжҲҗжіЁеҶҢ+зҷ»еҪ•гҖӮ вҖңи®ҫеӨҮе”ҜдёҖз»‘е®ҡвҖқжҳҜйЈҺжҺ§зӯ–з•ҘпјҲзү№еҲ«йҖӮеҗҲдҪ зҡ„вҖңд»…еҶ…йғЁе‘ҳе·ҘвҖқеңәжҷҜпјүпјҢдёҚжҳҜйҖҡиЎҢеҜҶй’Ҙзҡ„ејәеҲ¶иҰҒжұӮгҖӮ

еҸӘз»ҷйңҖиҰҒдҝ®ж”№/ж–°еўһзҡ„д»Јз ҒпјҲеӨҚжқӮйҖ»иҫ‘дёҠдёҖиЎҢжңүжіЁйҮҠпјү вҖӢ

A) PasskeysService.issueAuthenticationOptionsпјҡиӢҘеҸ‘зҺ°и®ҫеӨҮе·Із»‘е®ҡиҙҰеҸ·пјҢеҲҷе®ҡеҗ‘зҷ»еҪ•пјҲеҠ йҖҹжҲҗеҠҹзҺҮпјү вҖӢ

ts
// еӨҚжқӮйҖ»иҫ‘пјҡиӢҘ deviceIdentifier е·Із»‘е®ҡ userIdпјҢеҲҷе®ҡеҗ‘дёӢеҸ‘ allowCredentialsпјҲе‘ҪдёӯзҺҮжӣҙй«ҳпјҢжӣҙеҝ«пјү
async issueAuthenticationOptions({ deviceIdentifier }: { deviceIdentifier?: string }) {
  const relyingPartyId = this.configService.get<string>('RP_ID');

  // еҸҜйҖүзҡ„йҖҹзҺҮйҷҗеҲ¶пјҲ60s/5ж¬Ўпјү
  if (deviceIdentifier) {
    const rlKey = `rl:authpub:${deviceIdentifier}`;
    const c = await this.redisService.incrementKey(rlKey);
    if (c === 1) await this.redisService.expireKey(rlKey, 60);
    if (c > 5) throw new BadRequestException('иҜ·жұӮиҝҮдәҺйў‘з№ҒпјҢиҜ·зЁҚеҗҺйҮҚиҜ•');
  }

  let allowCredentials: { id: Buffer; type: 'public-key' }[] | undefined;
  let preferUserId: string | null = null;

  if (deviceIdentifier) {
    // еӨҚжқӮйҖ»иҫ‘пјҡеҰӮжһңиҜҘи®ҫеӨҮе·Із»‘е®ҡиҝҮиҙҰеҸ·пјҢдјҳе…Ҳиө°вҖңе®ҡеҗ‘зҷ»еҪ•вҖқ
    preferUserId = await this.redisService.getValue(`dvc:owner:${deviceIdentifier}`);
    if (preferUserId) {
      // еӨҚжқӮйҖ»иҫ‘пјҡжҹҘиҜўиҜҘз”ЁжҲ·зҡ„жүҖжңүеҮӯиҜҒпјҢжӢјеҮә allowCredentialsпјҲжҢүдҪ зҡ„жҢҒд№…еұӮжҺҘеҸЈи°ғж•ҙпјү
      const creds = await this.staffService.findManyCredentialsByUserId(preferUserId);
      allowCredentials = (creds || []).map((c: any) => ({
        id: Buffer.from(c.credentialId, 'base64url'),
        type: 'public-key' as const,
      }));
    }
  }

  const opts = await generateAuthenticationOptions({
    rpID: relyingPartyId,
    userVerification: 'preferred',
    // еӨҚжқӮйҖ»иҫ‘пјҡеӯҳеңЁ allowCredentials вҶ’ е®ҡеҗ‘зҷ»еҪ•пјӣеҗҰеҲҷз•ҷз©ә = еҸҜеҸ‘зҺ°еҮӯиҜҒ
    allowCredentials,
  });

  // еӨҚжқӮйҖ»иҫ‘пјҡжҢ‘жҲҳиҪ¬дёә base64urlпјҢеҲҶеҲ«жҢүвҖңuserIdвҖқжҲ–вҖңchallengeвҖқеҒҡж ЎйӘҢй”®
  const ch = typeof opts.challenge === 'string'
    ? opts.challenge
    : Buffer.from(opts.challenge).toString('base64url');

  if (preferUserId) {
    // е®ҡеҗ‘зҷ»еҪ•пјҡжҢү userId еӯҳ
    await this.redisService.setWithTimeToLive(`webauthn:auth:${preferUserId}`, ch, 300);
  } else {
    // ж— з”ЁжҲ·еҗҚзӣҙзҷ»пјҡжҢү challenge еӯҳ
    await this.redisService.setWithTimeToLive(`webauthn:auth:ch:${ch}`, '1', 300);
  }

  // е…је®№еүҚз«Ҝпјҡж— йңҖж”№еҚҸи®®пјҢд»Қиҝ”еӣһзәҜ AuthenticationOptions
  return { options: opts };
}

иҜҙжҳҺпјҡ

  • иҝҷж®өйҖ»иҫ‘дёҚж”№еҸҳдҪ зҡ„и·Ҝз”ұдёҺеүҚз«ҜеҚҸи®®пјҢд»…еңЁвҖңе·Із»‘е®ҡи®ҫеӨҮвҖқзҡ„жғ…еҶөдёӢз»ҷеҮә allowCredentialsпјҢжҲҗеҠҹзҺҮдјҡжӣҙй«ҳпјҢдё”д»Қдҝқз•ҷвҖңеҸҜеҸ‘зҺ°еҮӯиҜҒвҖқзҡ„й»ҳи®Өи·Ҝеҫ„гҖӮ
  • findManyCredentialsByUserId иҜ·еңЁдҪ зҡ„ AdminStaffService жҲ–еҮӯжҚ®д»“еә“йҮҢиЎҘдёҖдёӘз®ҖеҚ•жҹҘиҜўпјҲеҸӘиҰҒиҝ”еӣһ credentialId еҲ—иЎЁеҚіеҸҜпјүгҖӮ

B) еүҚз«ҜпјҲдҪ еҲҡеҶҷзҡ„ <script setup> йЎөйқўпјүзҷ»еҪ•жҢүй’®еўһеҠ вҖңеӨұиҙҘеӣһйҖҖеҲ°жіЁеҶҢвҖқзҡ„е…ңеә•пјҲеҸӘз»ҷж”№еҠЁж®өпјү вҖӢ

ts
// еӨҚжқӮйҖ»иҫ‘пјҡзҷ»еҪ•еӨұиҙҘж—¶иҮӘеҠЁеӣһйҖҖеҲ° JIT жіЁеҶҢпјҲжіЁеҶҢжҲҗеҠҹеҚізҷ»еҪ•пјү
async function onLogin() {
  try {
    appendLog("иҜ·жұӮзҷ»еҪ• options...");
    const { options } = await post("/passkeys/login/options", {
      deviceIdentifier: deviceId.value,
    });
    const requestOptionsJson = JSON.stringify(options);

    // еӨҚжқӮйҖ»иҫ‘пјҡе…Ҳе°қиҜ•зҷ»еҪ•пјҲAndroid ж”ҜжҢҒ conditional UIпјӣiOS еҝҪз•ҘиҜҘеҸӮж•°пјү
    const authRespJson = await passkeys.getPasskey(requestOptionsJson, {
      conditional: true,
    });

    appendLog("зҷ»еҪ•е“Қеә”иҝ”еӣһпјҢжҸҗдәӨеҗҺз«Ҝ verify...");
    const verify = await post(
      "/passkeys/login/verify",
      JSON.parse(authRespJson)
    );
    appendLog("зҷ»еҪ•е®ҢжҲҗпјҡ" + JSON.stringify(verify));
    uni.showToast({ title: "зҷ»еҪ•жҲҗеҠҹ", icon: "success" });
  } catch (e: any) {
    appendLog("жңӘжүҫеҲ°еҸҜз”ЁйҖҡиЎҢеҜҶй’ҘжҲ–з”ЁжҲ·еҸ–ж¶ҲпјҢиҪ¬е…ҘдёҖй”®жіЁеҶҢ...");
    // ===== еӣһйҖҖеҲ° JIT жіЁеҶҢ =====
    const { userId, options } = await post(
      "/passkeys/register-or-login/options",
      deviceId.value ? { deviceIdentifier: deviceId.value } : {}
    );
    const creationOptionsJson = JSON.stringify(options);
    const regRespJson = await passkeys.createPasskey(creationOptionsJson);

    appendLog("жіЁеҶҢе“Қеә”иҝ”еӣһпјҢжҸҗдәӨеҗҺз«Ҝ verify...");
    const verify = await post("/passkeys/register-or-login/verify", {
      ...JSON.parse(regRespJson),
      userId,
      deviceIdentifier: deviceId.value,
    });
    appendLog("жіЁеҶҢе№¶зҷ»еҪ•е®ҢжҲҗпјҡ" + JSON.stringify(verify));
    uni.showToast({ title: "е·ІжіЁеҶҢе№¶зҷ»еҪ•", icon: "success" });
  }
}
