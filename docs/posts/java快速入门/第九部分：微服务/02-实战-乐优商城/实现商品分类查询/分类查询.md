packagecom.leyou.item.service;

importcom.leyou.common.enums.ExceptionEnum;
importcom.leyou.common.exception.LyException;
importcom.leyou.item.mapper.CategoryMapper;
importcom.leyou.item.pojo.Category;
importorg.springframework.beans.factory.annotation.Autowired;
importorg.springframework.stereotype.Service;
importorg.springframework.util.CollectionUtils;

importjava.util.List;

@Service
public  class  CategoryService{
@Autowired
private CategoryMapper categoryMapper;

public List\<Category\> queryCategoryListByPid(Longpid){
Categoryt=new Category();
t.setParentId(pid);
//接收一个对象的非空属性作为查询条件
List\<Category\> list=categoryMapper.select(t);
//判断结果
if(CollectionUtils._isEmpty_(list)){
thrownew  LyException(ExceptionEnum._CATEGORY_NOT_FOUND_);
}
return  list;
}
}
