public void pushToConversionQueue(Long fileId) throws SchedulerException {
// 使用Quartz的JobBuilder创建一个新的JobDetail实例。这代表了一个Job的实例。
// 这里指定了Job的实现类为FileConversionJob，并给该Job设置了一个唯一的标识。
JobDetail jobDetail = JobBuilder.newJob(FileConversionJob.class)
.withIdentity("fileConversionJob" + fileId, "fileConversionGroup")
.build();
// 获取Job的数据映射，并将文件ID放入其中。这样，当Job执行时，我们可以从数据映射中获取到这个ID。
jobDetail.getJobDataMap().put("fileId", fileId);
// 使用Quartz的TriggerBuilder创建一个新的SimpleTrigger实例。
// 这里指定了Trigger的标识，并设置该Trigger立即启动。
// 还设置了Trigger的调度策略，即每10秒重复执行一次，并且无限重复。
SimpleTrigger trigger = TriggerBuilder.newTrigger()
.withIdentity("fileConversionTrigger" + fileId, "fileConversionGroup")
.startNow()
.withSchedule(SimpleScheduleBuilder.simpleSchedule()
.withIntervalInSeconds(10)
.repeatForever())
.build();
// 使用Scheduler（这是Quartz的核心组件）来安排Job的执行。
// 一旦Job被安排，它将根据上面定义的Trigger的策略被执行。
scheduler.scheduleJob(jobDetail, trigger);
}
