// 定义为Spring组件
@Component
// 启用Spring的配置属性功能，该功能会根据properties文件或YAML文件中的配置自动填充Bean的字段
@EnableConfigurationProperties(FileProperties.class)
// 注入需要的对象
@RequiredArgsConstructor(onConstructor = @__(@Autowired))
public class UploadUtil {

private static final Logger log = LoggerFactory.getLogger(UploadUtil.class);
private final FileProperties fileProperties;

private static FileProperties configFileProperties;

private static final String PATH_SEPARATOR = File.separator;
private static final DateTimeFormatter YEAR_MONTH_FORMATTER = DateTimeFormatter.ofPattern("yyyy/MM");

// 在Spring构造该Bean后执行这个方法
@PostConstruct
public void init() {
// 将Spring管理的Bean的引用赋值给静态变量，以便在静态方法中使用
configFileProperties = this.fileProperties;
}

// 上传文件的方法
public static FileVo upload(MultipartFile file, String moduleName, String childDir, Integer type) {
// 获取当前操作系统
String os = System.getProperty("os.name").toLowerCase();

// 根据不同的操作系统，设置不同的目录
String baseDir = "/home";
if (os.contains("mac")) {
baseDir = "/Users";
}

// 针对macOS, 我们需要更改配置文件中存储的路径为/Users/username/路径
String path;
if (os.contains("mac")) {
path = baseDir + "/" + System.getProperty("user.name") + configFileProperties.getPath();
} else {
path = configFileProperties.getPath();
}

// 获取当前日期的年份和月份字符串
String yearMonth = LocalDateTime.now().format(YEAR_MONTH_FORMATTER);

// 使用路径分隔符连接模块名、子目录和年份月份字符串，生成目录名后缀
String suffix = String.join(PATH_SEPARATOR,
moduleName.toLowerCase(),
childDir.toLowerCase(),
yearMonth);

// 使用Paths类的get方法拼接完整的目录路径
Path dirPath = Paths.get(path, suffix);

// 获取上传文件的名称，不包含扩展名
String fileName = FilenameUtils.getBaseName(file.getOriginalFilename());

// 获取上传文件的扩展名
String extension = FilenameUtils.getExtension(file.getOriginalFilename());

// 校验文件类型
if (extension == null || !EnumUtils.isValidEnumIgnoreCase(FilePatternEnum.class, extension)) {
throw new CustomerException(500, "文件格式错误!");
}

// 创建目录
File dir = dirPath.toFile();
if (!dir.exists() && !dir.mkdirs()) {
throw new CustomerException(500, "无法创建目录!");
}

// 检查文件是否为空
if (!file.isEmpty()) {
try {
// 使用Paths类的get方法拼接完整的文件路径
Path filePath = Paths.get(dirPath.toString(), UuidUtil.get32Uuid() + "." + extension);
File newFile = filePath.toFile();

// 转移文件
file.transferTo(newFile);

// 设置文件权限
if (!(newFile.setReadable(true) && newFile.setWritable(true))) {
log.warn("Failed to set readable/writable permissions for {}", newFile.getName());
}

// 构建并返回文件信息对象
return new FileVo().builder()
.fileName(fileName)
.filePath(suffix + PATH_SEPARATOR + newFile.getName())
.fileType(extension.toUpperCase())
.fileSize(FileUtils.byteCountToDisplaySize(newFile.length()))
.dataFlag(type)
.delFlag(LogicDelFlagEnum.UNDELETE.getValue())
.build();
} catch (IOException e) {
log.error("文件上传失败: ", e);
throw new CustomerException(500, "文件上传失败!");
}
} else {
throw new CustomerException(500, "未接收到文件,请重新尝试");
}
}

// 通过NIO方式给文件赋权，以保证文件能够正常打开
private static void changeFolderPermission(File file) throws IOException {
Set\<PosixFilePermission\> perms = new HashSet\<\>();
perms.add(PosixFilePermission.OWNER_READ);
perms.add(PosixFilePermission.OWNER_WRITE);
perms.add(PosixFilePermission.OWNER_EXECUTE);
perms.add(PosixFilePermission.GROUP_READ);
perms.add(PosixFilePermission.GROUP_WRITE);
perms.add(PosixFilePermission.GROUP_EXECUTE);
perms.add(PosixFilePermission.OTHERS_READ);
perms.add(PosixFilePermission.OTHERS_WRITE);
perms.add(PosixFilePermission.OTHERS_EXECUTE);
try {
Path path = Paths.get(file.getAbsolutePath());
Files.setPosixFilePermissions(path, perms);
} catch (Exception e) {
log.error("文件赋值权限异常：", e);
}
}
}
